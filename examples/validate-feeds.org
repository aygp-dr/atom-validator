* Validate a Simple Feed

#+begin_src clojure
(def simple-feed
  {:id "urn:uuid:feed-1"
   :title "My Tech Blog"
   :updated "2026-06-19T12:00:00Z"
   :authors [{:name "Jane Developer"}]
   :entries [{:id "urn:uuid:entry-1"
              :title "Getting Started with Clojure"
              :updated "2026-06-19T10:00:00Z"
              :links [{:href "https://example.com/clojure-intro/" :rel "alternate"}]}]})

(let [result (v/validate-feed simple-feed)]
  (println "Valid?" (:valid? result))
  (println "Errors:" (count (:errors result)))
  (println "Warnings:" (count (:warnings result))))
#+end_src

* Day-of-Week Mismatch (Issue #1)

This entry claims to be from "Thursday, June 19" but 2026-06-19 is a Friday.

#+begin_src clojure
(def mismatched-entry
  {:id "urn:uuid:brief-1"
   :title "Morning Brief: Thursday, June 19"  ; Wrong!
   :updated "2026-06-19T08:00:00Z"             ; This is Friday
   :links [{:href "https://news.example.com/brief-1/" :rel "alternate"}]})

(let [result (v/validate-entry mismatched-entry)]
  (println "Valid?" (:valid? result))
  (doseq [err (:errors result)]
    (println "  -" (:code err) "-" (:message err))))
#+end_src

Expected output:
#+begin_example
Valid? false
  - :day-of-week-mismatch - Title says 'Thursday' but updated date is a Friday
#+end_example

* Stale Feed Updated (Issue #2)

The feed's =<updated>= timestamp must be >= the newest entry's timestamp.

#+begin_src clojure
(def stale-feed
  {:id "urn:uuid:feed-2"

              :title "Breaking News"
              :updated "2026-06-19T15:00:00Z"  ; Newer than feed!
              :links [{:href "https://news.example.com/breaking/" :rel "alternate"}]}]})

(let [result (v/validate-feed stale-feed)]

* Fetch and Validate a Real Feed

#+begin_src clojure
(defn fetch-and-validate [url]
  (try
    (let [xml (slurp url)
          result (v/validate-feed xml)]
      (println "Feed:" url)
      (println "Valid?" (:valid? result))
      (println "Entries:" (count (get-in result [:feed :entries])))
      (when (seq (:errors result))
        (println "Errors:")
        (doseq [err (take 5 (:errors result))]
          (println "  -" (:code err))))
      result)
    (catch Exception e
      (println "Error fetching:" (.getMessage e))
      nil)))

;; Validate GitHub's Clojure releases feed
(fetch-and-validate "https://github.com/clojure/clojure/releases.atom")
#+end_src

* Batch Validation

Validate multiple feeds and collect statistics.

#+begin_src clojure
(def feed-urls
  ["https://github.com/clojure/clojure/releases.atom"
   "https://github.com/babashka/babashka/releases.atom"
   "https://github.com/cognitect-labs/test-runner/releases.atom"])

(defn validate-batch [urls]
  (let [results (for [url urls]
                  (try
                    {:url url
                     :result (v/validate-feed (slurp url))}
                    (catch Exception e
                      {:url url
                       :error (.getMessage e)})))]
    (println "Batch validation results:")
    (println "  Total:" (count results))
    (println "  Valid:" (count (filter #(get-in % [:result :valid?]) results)))
    (println "  Invalid:" (count (filter #(and (:result %) (not (get-in % [:result :valid?]))) results)))
    (println "  Errors:" (count (filter :error results)))
    results))

(validate-batch feed-urls)
#+end_src
